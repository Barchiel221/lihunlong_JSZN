### B.1 IMU 前向传播与点云反向去畸变 [`src/FAST_LIO/src/IMU_Processing.hpp`](src/FAST_LIO/src/IMU_Processing.hpp)

```cpp
void ImuProcess::UndistortPcl(const MeasureGroup &meas,
                              esekfom::esekf<state_ikfom, 12, input_ikfom> &kf_state,
                              PointCloudXYZI &pcl_out)
{
  // ---------- 1. 把上一帧尾部 IMU 加到本帧头部，保证 dt 连续 ----------
  auto v_imu = meas.imu;
  v_imu.push_front(last_imu_);
  const double &pcl_beg_time = meas.lidar_beg_time;
  const double &pcl_end_time = meas.lidar_end_time;

  pcl_out = *(meas.lidar);
  sort(pcl_out.points.begin(), pcl_out.points.end(), time_list);  // 按每点 offset 排序

  // ---------- 2. 前向递推 IESKF：在每段 IMU 间隔上 predict ----------
  state_ikfom imu_state = kf_state.get_x();
  IMUpose.clear();
  IMUpose.push_back(set_pose6d(0.0, acc_s_last, angvel_last,
                               imu_state.vel, imu_state.pos,
                               imu_state.rot.toRotationMatrix()));

  V3D angvel_avr, acc_avr;  double dt = 0;  input_ikfom in;
  for (auto it = v_imu.begin(); it < v_imu.end() - 1; ++it) {
    auto &head = *it, &tail = *(it + 1);
    angvel_avr = 0.5 * (toEigen(head->angular_velocity)   + toEigen(tail->angular_velocity));
    acc_avr    = 0.5 * (toEigen(head->linear_acceleration) + toEigen(tail->linear_acceleration));
    acc_avr    = acc_avr * G_m_s2 / mean_acc.norm();          // 重力对齐归一化

    dt = rclcpp::Time(tail->header.stamp).seconds()
       - rclcpp::Time(head->header.stamp).seconds();

    in.acc = acc_avr;  in.gyro = angvel_avr;
    Q.block<3,3>(0,0).diagonal() = cov_gyr;
    Q.block<3,3>(3,3).diagonal() = cov_acc;
    Q.block<3,3>(6,6).diagonal() = cov_bias_gyr;
    Q.block<3,3>(9,9).diagonal() = cov_bias_acc;
    kf_state.predict(dt, Q, in);                              // 标称 + 协方差 一阶递推

    imu_state   = kf_state.get_x();
    angvel_last = angvel_avr - imu_state.bg;
    acc_s_last  = imu_state.rot * (acc_avr - imu_state.ba) + imu_state.grav;

    double offs_t = rclcpp::Time(tail->header.stamp).seconds() - pcl_beg_time;
    IMUpose.push_back(set_pose6d(offs_t, acc_s_last, angvel_last,
                                 imu_state.vel, imu_state.pos,
                                 imu_state.rot.toRotationMatrix()));
  }

  // ---------- 3. 反向传播：把每个 LiDAR 点补偿到"帧末"位姿 ----------
  // P_compensate = R_LI^T·(R_e^T·(R_i·(R_LI·P_i + t_LI) + T_ei) − t_LI)
  auto it_pcl = pcl_out.points.end() - 1;
  for (auto it_kp = IMUpose.end() - 1; it_kp != IMUpose.begin(); --it_kp) {
    auto head = it_kp - 1, tail = it_kp;
    M3D R_imu;  V3D vel_imu, pos_imu, acc_imu;
    R_imu  << MAT_FROM_ARRAY(head->rot);
    vel_imu<< VEC_FROM_ARRAY(head->vel);
    pos_imu<< VEC_FROM_ARRAY(head->pos);
    acc_imu<< VEC_FROM_ARRAY(tail->acc);
    angvel_avr << VEC_FROM_ARRAY(tail->gyr);

    for (; it_pcl->curvature / 1000.0 > head->offset_time; --it_pcl) {
      dt = it_pcl->curvature / 1000.0 - head->offset_time;
      M3D R_i = R_imu * Exp(angvel_avr, dt);
      V3D P_i(it_pcl->x, it_pcl->y, it_pcl->z);
      V3D T_ei = pos_imu + vel_imu*dt + 0.5*acc_imu*dt*dt - imu_state.pos;
      V3D P_c  = imu_state.offset_R_L_I.conjugate() *
                 (imu_state.rot.conjugate() *
                  (R_i*(imu_state.offset_R_L_I*P_i + imu_state.offset_T_L_I) + T_ei)
                  - imu_state.offset_T_L_I);
      it_pcl->x = P_c(0);  it_pcl->y = P_c(1);  it_pcl->z = P_c(2);
      if (it_pcl == pcl_out.points.begin()) break;
    }
  }
}
```

```cpp
bool EGOPlannerManager::reboundReplan(...) {
    bspline_optimizer_->setLocalTargetPt(local_target_pt);
    bspline_optimizer_->setAltitudeRef(start_pt(2));   // altitude penalty 参考高度

    // STEP 1 — 初值生成：一段 polynomial / minSnap 经过随机插入点
    double ts = (start_pt - local_target_pt).norm() > 0.1
              ? pp_.ctrl_pt_dist / pp_.max_vel_ * 1.5
              : pp_.ctrl_pt_dist / pp_.max_vel_ * 5;
    PolynomialTraj gl_traj;
    double dist = (start_pt - local_target_pt).norm();
    double time = pow(max_vel,2)/max_acc > dist
                ? sqrt(dist/max_acc)
                : (dist - pow(max_vel,2)/max_acc)/max_vel + 2*max_vel/max_acc;
    if (!flag_randomPolyTraj)
        gl_traj = PolynomialTraj::one_segment_traj_gen(start_pt, start_vel, start_acc,
                                                       local_target_pt, local_target_vel,
                                                       Vector3d::Zero(), time);
    else { /* minSnapTraj with a random horizontal/vertical perturbation */ }

    // 把 polynomial 沿 ts 等距采样为 point_set
    for (double t=0; t<time; t+=ts) point_set.push_back(gl_traj.evaluate(t));

    // STEP 1.5 — point_set → 均匀 B 样条控制点
    UniformBspline::parameterizeToBspline(ts, point_set, start_end_derivatives, ctrl_pts);

    bool flag_step_1_success =
        bspline_optimizer_->BsplineOptimizeTrajRebound(ctrl_pts, ts);  // -> C.3

    // STEP 3 — 时间重分配 + 二次平滑 (refine)
    UniformBspline pos = UniformBspline(ctrl_pts, 3, ts);
    pos.setPhysicalLimits(pp_.max_vel_, pp_.max_acc_, pp_.feasibility_tolerance_);
    double ratio;  bool feasible = pos.checkFeasibility(ratio, false);

    bspline_optimizer_->BsplineOptimizeTrajRefine(ctrl_pts, pos.getInterval(), ctrl_pts_temp);
```

```cpp
void BsplineOptimizer::calcSmoothnessCost(const MatrixXd &q, double &cost, MatrixXd &gradient,
                                          bool flag_use_jerk /*=true*/) {
    cost = 0.0;
    if (flag_use_jerk) {
        Vector3d jerk;
        for (int i = 0; i < q.cols()-3; i++) {
            jerk = q.col(i+3) - 3*q.col(i+2) + 3*q.col(i+1) - q.col(i);  // ≈ d³p/dt³ × ts³
            cost += jerk.squaredNorm();
            Vector3d temp = 2.0 * jerk;
            gradient.col(i+0) += -temp;     gradient.col(i+1) +=  3*temp;
            gradient.col(i+2) += -3*temp;   gradient.col(i+3) +=    temp;
        }
    }
}
```

#### C.3.2 ESDF-free 障碍距离代价（{p,v} 配对 + 三次罚函数）

```cpp
void BsplineOptimizer::calcDistanceCostRebound(const MatrixXd &q, double &cost,
                                               MatrixXd &gradient, int iter_num, double smoothness_cost) {
    int end_idx = q.cols() - order_;
    double demarcation = cps_.clearance;
    double a = 3*demarcation, b = -3*pow(demarcation,2), c = pow(demarcation,3);

    // 平滑度足够后才允许 rebound（避免初值劣化时反弹方向乱跳）
    if (iter_num > 3 && smoothness_cost/(cps_.size - 2*order_) < 0.1)
        check_collision_and_rebound();

    for (auto i = order_; i < end_idx; ++i) {
        for (size_t j = 0; j < cps_.direction[i].size(); ++j) {
            double dist     = (cps_.points.col(i) - cps_.base_point[i][j]).dot(cps_.direction[i][j]);
            double dist_err = cps_.clearance - dist;
            Vector3d dist_grad = cps_.direction[i][j];
            if (dist_err < 0)                continue;
            else if (dist_err < demarcation) {                       // 三次段
                cost += pow(dist_err, 3);
                gradient.col(i) += -3.0 * dist_err*dist_err * dist_grad;
            } else {                                                 // 大偏离切到二次段保数值稳定
                cost += a*dist_err*dist_err + b*dist_err + c;
                gradient.col(i) += -(2*a*dist_err + b) * dist_grad;
            }
        }
    }
}
```

#### C.3.3 高度惩罚（本工作新增，针对实飞 z 漂移）

```cpp
// 对每个内部控制点惩罚 (z - z_ref)^2,
// 让 −Z 下钻 / +Z 抬升都被定价, 优化器更偏好水平 ±Y 侧绕。
void BsplineOptimizer::calcAltitudeCost(const MatrixXd &q, double &cost, MatrixXd &gradient) {
    cost = 0.0;
    int end_idx = q.cols() - order_;
    for (int i = order_; i < end_idx; ++i) {
        double dz = q(2, i) - z_ref_;
        cost += dz * dz;
        gradient(2, i) += 2.0 * dz;
    }
}
```

#### C.3.4 动力学可行性（vel / acc 双向超限三次罚）

```cpp
void BsplineOptimizer::calcFeasibilityCost(const MatrixXd &q, double &cost, MatrixXd &gradient) {
    double ts = bspline_interval_, ts_inv2 = 1/ts/ts, ts_inv3 = 1/ts/ts/ts;
    /* velocity: vi = (q_{i+1}-q_i)/ts */
    for (int i = 0; i < q.cols()-1; i++) {
        Vector3d vi = (q.col(i+1) - q.col(i)) / ts;
        for (int j = 0; j < 3; j++) {
            if (vi(j) > max_vel_) {
                double diff = vi(j) - max_vel_;
                cost += pow(diff, 3) * ts_inv3;
                double grad = 3 * diff * diff / ts * ts_inv3;
                gradient(j, i+0) += -grad;  gradient(j, i+1) += grad;
            } else if (vi(j) < -max_vel_) { /* 对称分支 */ }
        }
    }
    /* acceleration: ai = (q_{i+2}-2 q_{i+1}+q_i)/ts² ; 三次罚结构同上 */
}
```

```cpp
void BsplineOptimizer::combineCostRebound(const double *x, double *grad, double &f_combine, const int n) {
    memcpy(cps_.points.data() + 3*order_, x, n * sizeof(x[0]));
    double f_smoothness, f_distance, f_feasibility, f_swarm, f_terminal, f_altitude;
    MatrixXd g_smoothness, g_distance, g_feasibility, g_swarm, g_terminal, g_altitude;
    g_smoothness = g_distance = g_feasibility = g_swarm = g_terminal = g_altitude
                 = MatrixXd::Zero(3, cps_.size);

    calcSmoothnessCost   (cps_.points, f_smoothness,   g_smoothness);
    calcDistanceCostRebound(cps_.points, f_distance,   g_distance, iter_num_, f_smoothness);
    calcFeasibilityCost  (cps_.points, f_feasibility,  g_feasibility);
    calcSwarmCost        (cps_.points, f_swarm,        g_swarm);
    calcTerminalCost     (cps_.points, f_terminal,     g_terminal);
    calcAltitudeCost     (cps_.points, f_altitude,     g_altitude);

    f_combine =  lambda1_ * f_smoothness
              +  new_lambda2_ * f_distance
              +  lambda3_ * f_feasibility
              +  new_lambda2_ * f_swarm
              +  lambda2_ * f_terminal
              +  lambda_z_ * f_altitude;

    MatrixXd grad_3D = lambda1_ * g_smoothness + new_lambda2_ * g_distance
                     + lambda3_ * g_feasibility + new_lambda2_ * g_swarm
                     + lambda2_ * g_terminal    + lambda_z_   * g_altitude;
    memcpy(grad, grad_3D.data() + 3*order_, n * sizeof(grad[0]));
}
```

### D.1 LIO → PX4 EKF2 视觉里程计桥 [`src/lio_px4_bridge/src/bridge_node.cpp`](src/lio_px4_bridge/src/bridge_node.cpp)

> 作用：把 FAST-LIO 输出的 10 Hz `/Odometry` (ENU/FLU) 升频到 200 Hz `/fmu/in/vehicle_visual_odometry` (NED/FRD)，
> 通过 IMU strapdown 短程积分填补 LIO 帧间，并以 `/cloud_effected` 点数做退化检测放大协方差。

```cpp
// map → NED 绕 X 轴 180°; FLU → FRD 同理
const Eigen::Quaterniond kQ_MAP2NED(0,1,0,0);
const Eigen::Quaterniond kQ_FLU2FRD(0,1,0,0);
constexpr double kGravity = 9.80665;

// SO(3) 指数映射 (一阶近似 + 精确公式)
inline Eigen::Quaterniond expSO3(const Eigen::Vector3d& w) {
  const double n = w.norm();
  if (n < 1e-9) return Eigen::Quaterniond(1, 0.5*w.x(), 0.5*w.y(), 0.5*w.z()).normalized();
  const double h = 0.5*n, s = std::sin(h)/n;
  return Eigen::Quaterniond(std::cos(h), s*w.x(), s*w.y(), s*w.z());
}

void on_lio_odom(const nav_msgs::msg::Odometry::SharedPtr m) {
  std::lock_guard<std::mutex> lk(mtx_);
  const Eigen::Vector3d p_map(m->pose.pose.position.x, m->pose.pose.position.y, m->pose.pose.position.z);
  const Eigen::Quaterniond q_map_flu(m->pose.pose.orientation.w, m->pose.pose.orientation.x,
                                     m->pose.pose.orientation.y, m->pose.pose.orientation.z);
  // 二点差分速度种子 (FAST-LIO 自带 twist 为 0, 不可信)
  Eigen::Vector3d v_seed = Eigen::Vector3d::Zero();
  const rclcpp::Time t_now = m->header.stamp;
  if (have_last_lio_) {
    double dt = (t_now - last_lio_t_).seconds();
    if (dt > 1e-3 && dt < 1.0) v_seed = (p_map - last_lio_p_map_) / dt;
  }
  last_lio_t_ = t_now;  last_lio_p_map_ = p_map;  have_last_lio_ = true;

  anchor_p_ned_     = kQ_MAP2NED * p_map;
  anchor_q_ned_frd_ = (kQ_MAP2NED * q_map_flu * kQ_FLU2FRD.conjugate()).normalized();
  anchor_v_ned_     = kQ_MAP2NED * v_seed;
  anchor_t_         = t_now;
  anchor_ready_     = true;
}

void on_imu(const sensor_msgs::msg::Imu::SharedPtr m) {
  // Livox SDK 给 sensor_msgs/Imu 时 acc 单位是 g, 需 ×9.80665 转 m/s²
  const Eigen::Vector3d acc_frd(   m->linear_acceleration.x,
                                 - m->linear_acceleration.y,
                                 - m->linear_acceleration.z );
  const Eigen::Vector3d acc_si  = acc_frd * kGravity;
  const Eigen::Vector3d gyr_frd(   m->angular_velocity.x,
                                 - m->angular_velocity.y,
                                 - m->angular_velocity.z );
  last_gyr_frd_ = gyr_frd;
  if (!anchor_ready_.load()) { last_imu_t_=t_now; have_last_imu_=true; return; }

  double dt = (t_now - last_imu_t_).seconds();
  last_imu_t_ = t_now;
  if (dt <= 0) return;  if (dt > imu_max_dt_) dt = imu_max_dt_;  // 丢包保护

  // strapdown: f_FRD = a_body_inertial - g_body  ⇒  a_world = R · f_body + g_world
  const Eigen::Matrix3d R = anchor_q_ned_frd_.toRotationMatrix();
  const Eigen::Vector3d g_ned(0, 0, kGravity);
  const Eigen::Vector3d a_world = R * acc_si + g_ned;
  anchor_v_ned_     += a_world * dt;
  anchor_p_ned_     += anchor_v_ned_ * dt;
  anchor_q_ned_frd_  = (anchor_q_ned_frd_ * expSO3(gyr_frd * dt)).normalized();
  anchor_t_          = t_now;
  if (publish_on_imu_) publish_vo();
}

void on_effect_cloud(const sensor_msgs::msg::PointCloud2::SharedPtr m) {
  // 有效配对点 < 100 → 协方差 ×100 通知 EKF2 暂时降权 LIO
  const uint32_t n = m->width * m->height;
  cov_scale_ = (static_cast<int>(n) < degraded_threshold_) ? degraded_cov_scale_ : 1.0;
}

void publish_vo() {
  px4_msgs::msg::VehicleOdometry vo{};
  vo.timestamp        = steady_clock_us();
  vo.timestamp_sample = static_cast<uint64_t>(anchor_t_.nanoseconds() / 1000);
  vo.pose_frame       = vo.POSE_FRAME_NED;
  vo.velocity_frame   = vo.VELOCITY_FRAME_BODY_FRD;

  // 位姿 (NED, FRD)
  vo.position = { (float)anchor_p_ned_.x(),     (float)anchor_p_ned_.y(),     (float)anchor_p_ned_.z()     };
  vo.q        = { (float)anchor_q_ned_frd_.w(), (float)anchor_q_ned_frd_.x(),
                  (float)anchor_q_ned_frd_.y(), (float)anchor_q_ned_frd_.z() };

  // 速度: NED 世界系 → body FRD
  const Eigen::Vector3d v_body = anchor_q_ned_frd_.conjugate() * anchor_v_ned_;
  vo.velocity = { (float)v_body.x(), (float)v_body.y(), (float)v_body.z() };
  vo.angular_velocity = { (float)last_gyr_frd_.x(), (float)last_gyr_frd_.y(), (float)last_gyr_frd_.z() };

  const float scale = (float)cov_scale_.load();
  for (int i = 0; i < 3; ++i) {
    vo.position_variance[i]    = (float)pos_var_  * scale;
    vo.orientation_variance[i] = (float)ori_var_  * scale;
    vo.velocity_variance[i]    = (float)vel_var_  * scale;
  }
  pub_vo_->publish(vo);
}
```

### D.2 EGO PositionCommand → PX4 TrajectorySetpoint 适配器 [`src/ego_px4_adapter/src/ego_px4_adapter_node.cpp`](src/ego_px4_adapter/src/ego_px4_adapter_node.cpp)

> 作用：把 `traj_server` 出的 `quadrotor_msgs/PositionCommand` (map=ENU/FLU)
> 转换为 PX4 `trajectory_setpoint` (NED)，仅做坐标系翻 `y/z` 符号 + yaw 取反。

```cpp
void EgoPx4Adapter::cb(const quadrotor_msgs::msg::PositionCommand::SharedPtr msg) {
    px4_msgs::msg::TrajectorySetpoint sp{};
    sp.timestamp = static_cast<uint64_t>(now().nanoseconds() / 1000ULL);

    // Position: map(x N, y W, z U) → NED(x N, y E, z D)
    sp.position[0] = (float)  msg->position.x;
    sp.position[1] = (float)(-msg->position.y);
    sp.position[2] = (float)(-msg->position.z);

    if (pub_vel_) {
        sp.velocity[0] = (float)  msg->velocity.x;
        sp.velocity[1] = (float)(-msg->velocity.y);
        sp.velocity[2] = (float)(-msg->velocity.z);
    } else sp.velocity.fill(NAN);

    if (pub_acc_) {
        sp.acceleration[0] = (float)  msg->acceleration.x;
        sp.acceleration[1] = (float)(-msg->acceleration.y);
        sp.acceleration[2] = (float)(-msg->acceleration.z);
    } else sp.acceleration.fill(NAN);

    sp.jerk.fill(NAN);

    // Yaw: 两系差一个绕 X 轴 180° → yaw_ned = -yaw_map, yawrate 同理
    sp.yaw      = wrap_pi((float)(-msg->yaw));
    sp.yawspeed =       (float)(-msg->yaw_dot);
    pub_->publish(sp);
}
```

### D.3 PX4 Offboard 任务状态机（解锁 / 起飞 / EGO 巡航 / MID360 二次校准 / 自动降落）[`src/px4_offboard_pkg/src/offboard_node.cpp`](src/px4_offboard_pkg/src/offboard_node.cpp)

> 11 个状态：`INIT → WARMUP → ENGAGE → ARMING → TAKEOFF → FLY → FINAL_CORRECTION → HOLD → LANDING → DONE/ABORT`。
> 安全设计：默认 `enable_arm=false` 仅发 setpoint 不解锁；任何状态 30 s 超时 / 失保 / 用户接管立即转 `LANDING/ABORT`。

```cpp
enum class State { INIT, WARMUP, ENGAGE, ARMING, TAKEOFF, FLY,
                   FINAL_CORRECTION, HOLD, LANDING, DONE, ABORT };

void on_timer() {
  publish_offboard_control();   // 每 tick 必须发 OffboardControlMode (>2 Hz 否则 PX4 失保)

  // 全局保护
  if (state_ != INIT && state_ != WARMUP && state_ != DONE && state_ != ABORT) {
    if (any_failsafe())            { cmd_land();          set_state(LANDING, "failsafe"); }
    else if (user_took_over())     {                      set_state(ABORT,   "user override"); }
    else if (seconds_in_state() > max_state_sec_) {
                                     cmd_land();          set_state(LANDING, "timeout"); }
  }

  switch (state_) {
  case INIT:
    if (pos_valid_ && status_recv_) {
      home_x_ = pos_.x;  home_y_ = pos_.y;  home_yaw_ = pos_.heading;     // 锚 home
      publish_setpoint(home_x_, home_y_, -takeoff_alt_, home_yaw_);
      set_state(WARMUP, "pos valid, capturing home");
    } else publish_setpoint(0, 0, -0.1, 0);
    break;

  case WARMUP:                          // 灌 1~2s setpoint 流，满足 PX4 Offboard 前提
    publish_setpoint(home_x_, home_y_, -takeoff_alt_, home_yaw_);
    if (seconds_in_state() >= warmup_sec_) {
      if (!enable_arm_) set_state(HOLD, "dry-run park");
      else { cmd_set_offboard(); set_state(ENGAGE, "set Offboard"); }
    } break;

  case ENGAGE:
    publish_setpoint(home_x_, home_y_, -takeoff_alt_, home_yaw_);
    if (status_.nav_state == VehicleStatus::NAVIGATION_STATE_OFFBOARD) {
      cmd_arm();  set_state(ARMING, "armed cmd sent");
    } else if (seconds_in_state() > 5.0) { cmd_set_offboard(); /* retry */ }
    break;

  case ARMING:
    publish_setpoint(home_x_, home_y_, -takeoff_alt_, home_yaw_);
    if (status_.arming_state == VehicleStatus::ARMING_STATE_ARMED)
      set_state(TAKEOFF, "armed, climbing");
    else if (seconds_in_state() > 5.0) cmd_arm();
    break;

  case TAKEOFF:
    publish_setpoint(home_x_, home_y_, -takeoff_alt_, home_yaw_);
    if (reached(home_x_, home_y_, -takeoff_alt_, home_yaw_))
      set_state(FLY, "altitude reached");
    break;

  case FLY:
    if (enable_ext_traj_ && ext_sp_have_ && ext_sp_fresh()) {
      // 转发 EGO 轨迹 (位置+速度+加速度) 给 PX4, 但 yaw 强锁 home_yaw_:
      // traj_server 的 yaw 来自路径切向, 起步/重规划时容易跳到 ~90°
      auto sp = ext_sp_;  sp.timestamp = now_us();
      sp.yaw = (float)home_yaw_;  sp.yawspeed = NAN;
      pub_sp_->publish(sp);
      if (ego_traj_at_goal()) {                                  // EGO 末端到点
        hold_x_ = goal_map_x_;  hold_y_ = -goal_map_y_;  hold_z_ = -goal_map_z_;
        hold_yaw_ = home_yaw_;
        set_state(FINAL_CORRECTION, "ego goal reached, using MID360 odom");
      }
    } else {
      if (enable_ext_traj_ && !ext_sp_fresh())
          RCLCPP_WARN_THROTTLE(...,"External traj timeout, fallback to target");
      publish_setpoint(target_x_, target_y_, -takeoff_alt_, target_yaw_);
    } break;

  case FINAL_CORRECTION:               // EGO 收尾后, 用 MID360 直读 odom 做最终判定
    publish_hold_target();
    if (mid360_odom_fresh() && mid360_reached_goal())
      set_state(HOLD, "MID360 precision goal reached");
    break;

  case HOLD:
    publish_hold_target();
    if (enable_arm_ && seconds_in_state() >= hold_sec_) {
      cmd_land();  set_state(LANDING, "hold done");
    } break;

  case LANDING:
    publish_setpoint(hold_x_, hold_y_, pos_.z, pos_.heading);    // 兜底 setpoint
    if (auto_disarm_after_landing_ && land_detected_.landed &&
        seconds_in_state() >= land_disarm_delay_sec_) cmd_disarm();
    if (status_.arming_state == VehicleStatus::ARMING_STATE_DISARMED)
      set_state(DONE, "disarmed");
    break;
  }
}

// 单航点判定: xy 欧氏 + z 绝对 + yaw 环绕到 [-π, π]
bool reached(double tx, double ty, double tz, double tyaw) const {
  if (!pos_valid_) return false;
  if (std::hypot(tx-pos_.x, ty-pos_.y) > xy_tol_)  return false;
  if (std::fabs(tz - pos_.z) > z_tol_)             return false;
  double e = tyaw - pos_.heading;
  while (e>M_PI)  e -= 2*M_PI;
  while (e<-M_PI) e += 2*M_PI;
  return std::fabs(e) <= yaw_tol_;
}

// EGO 轨迹末端到点的"持续 N 秒"判定 (避免单帧抖到容差内即误触)
bool ego_traj_at_goal() {
  if (!goal_have_ || !ext_sp_have_ || !ext_sp_fresh()) { ego_goal_close_since_={}; return false; }
  const double ex = ext_sp_.position[0] - goal_map_x_;
  const double ey = ext_sp_.position[1] - (-goal_map_y_);
  const double ez = ext_sp_.position[2] - (-goal_map_z_);
  const bool close = std::hypot(ex,ey) <= ego_goal_xy_tol_ &&
                     std::fabs(ez)     <= ego_goal_z_tol_;
  if (!close) { ego_goal_close_since_ = {}; return false; }
  if (ego_goal_close_since_.time_since_epoch().count() == 0)
    { ego_goal_close_since_ = std::chrono::steady_clock::now(); return false; }
  return std::chrono::duration<double>(steady_clock::now() - ego_goal_close_since_).count()
         >= ego_goal_settle_sec_;
}
```

| 链路                                                       | 频率      | 数据类型                            | QoS                                |
| ---------------------------------------------------------- | --------- | ----------------------------------- | ---------------------------------- |
| `livox_ros_driver2 → /livox/lidar`                         | 10 Hz     | `livox_ros_driver2/CustomMsg`       | SensorDataQoS (best-effort)        |

| `FAST_LIO   → /Odometry`                                   | 10 Hz     | `nav_msgs/Odometry` (ENU)           | default reliable                   |
| `FAST_LIO   → /cloud_registered`                           | 10 Hz     | `sensor_msgs/PointCloud2` (ENU)     | default reliable                   |
| `lio_px4_bridge → /fmu/in/vehicle_visual_odometry`         | 200 Hz    | `px4_msgs/VehicleOdometry` (NED)    | KeepLast(1) best-effort volatile   |
| `ego_planner → /drone_0_planning/bspline`                  | 重规划时  | `traj_utils/Bspline`                | default reliable                   |
| `traj_server → /position_cmd`                              | 50 Hz     | `quadrotor_msgs/PositionCommand`    | default reliable                   |
| `ego_px4_adapter → /ego/trajectory_setpoint`               | 50 Hz     | `px4_msgs/TrajectorySetpoint` (NED) | default reliable                   |
| `px4_offboard_pkg → /fmu/in/trajectory_setpoint`           | 50 Hz     | `px4_msgs/TrajectorySetpoint`       | KeepLast(1) best-effort volatile   |
| `px4_offboard_pkg → /fmu/in/offboard_control_mode`         | 50 Hz     | `px4_msgs/OffboardControlMode`      | KeepLast(1) best-effort volatile   |
